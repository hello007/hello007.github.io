1. Chunking（分块策略）

🎯 掌握程度：必须掌握

什么是Chunking？

通俗解释：
Chunking就是"把长文章切成小段"。就像把长面包切成片，方便食用。把长文档切成小段，方便AI理解和检索。

技术定义：
将长文本分割成适合处理的小块（chunks），是构建向量数据库的核心步骤。

为什么需要Chunking？

问题：长文档无法直接处理

❌ 整个文档作为一个向量：
├─ 文档太长：超过Token限制
├─ 信息过载：检索时返回过多无关信息
├─ 语义模糊：向量无法准确表示
└─ 存储浪费：每个文档一个向量，空间利用率低

✅ 文档分块后：
├─ 精确检索：只返回相关段落
├─ 语义清晰：每块语义完整
├─ 存储高效：多个向量，灵活检索
└─ 成本降低：只检索需要的部分

Chunking策略对比

策略1：固定大小分块（Fixed-size）

@Service
public class FixedSizeChunker {

    private static final int CHUNK_SIZE = 500;  // 字符数
    private static final int OVERLAP = 50;      // 重叠字符数

    public List<String> chunk(String text) {

        List<String> chunks = new ArrayList<>();
        int start = 0;

        while (start < text.length()) {

            int end = Math.min(start + CHUNK_SIZE, text.length());

            // 提取chunk
            String chunk = text.substring(start, end);
            chunks.add(chunk);

            // 移动到下一个chunk（考虑重叠）
            start = end - OVERLAP;

            // 防止无限循环
            if (start >= text.length()) {
                break;
            }
        }

        return chunks;
    }
}

优点：

  • 实现简单
  • 性能好
  • 覆盖全面

缺点：

  • 可能在句子中间截断
  • 语义不完整

策略2：段落分块（Paragraph-based）

@Service
public class ParagraphChunker {

    public List<String> chunkByParagraph(String text) {

        // 按段落分割（\n\n 或 \n）
        String[] paragraphs = text.split("\\n\\n|\\n");

        List<String> chunks = new ArrayList<>();
        StringBuilder currentChunk = new StringBuilder();
        int currentLength = 0;
        final int MAX_LENGTH = 1000;

        for (String paragraph : paragraphs) {

            // 如果单个段落太长，需要进一步分割
            if (paragraph.length() > MAX_LENGTH) {
                if (currentChunk.length() > 0) {
                    chunks.add(currentChunk.toString());
                    currentChunk = new StringBuilder();
                    currentLength = 0;
                }

                // 对长段落进行强制分割
                List<String> subChunks = splitLongParagraph(paragraph, MAX_LENGTH);
                chunks.addAll(subChunks);
                continue;
            }

            // 检查是否超出限制
            if (currentLength + paragraph.length() > MAX_LENGTH) {
                if (currentChunk.length() > 0) {
                    chunks.add(currentChunk.toString());
                    currentChunk = new StringBuilder();
                    currentLength = 0;
                }
            }

            // 添加段落
            currentChunk.append(paragraph).append("\n\n");
            currentLength += paragraph.length() + 2;
        }

        // 添加最后一个chunk
        if (currentChunk.length() > 0) {
            chunks.add(currentChunk.toString());
        }

        return chunks;
    }

    private List<String> splitLongParagraph(String paragraph, int maxLength) {

        List<String> chunks = new ArrayList<>();
        int start = 0;

        while (start < paragraph.length()) {

            int end = Math.min(start + maxLength, paragraph.length());

            // 尝试在句子边界分割
            if (end < paragraph.length()) {
                // 往前找最近的句号
                int lastPeriod = paragraph.lastIndexOf("。", end);
                if (lastPeriod > start) {
                    end = lastPeriod + 1;
                }
            }

            chunks.add(paragraph.substring(start, end).trim());
            start = end;
        }

        return chunks;
    }
}

优点：

  • 保持段落完整
  • 语义相对完整
  • 更易阅读

缺点：

  • 实现复杂
  • 段落长度不均

策略3：语义分块（Semantic）

@Service
public class SemanticChunker {

    @Autowired
    private EmbeddingModel embeddingModel;

    public List<String> chunkBySemantic(String text) {

        // 1. 先按段落粗分
        List<String> paragraphs = Arrays.asList(text.split("\\n\\n"));

        // 2. 计算每段的向量
        List<float[]> embeddings = paragraphs.stream()
            .map(para -> embeddingModel.embed(para).content())
            .collect(Collectors.toList());

        // 3. 计算相邻段落的语义距离
        List<Double> distances = new ArrayList<>();
        for (int i = 0; i < embeddings.size() - 1; i++) {
            double distance = 1 - cosineSimilarity(
                embeddings.get(i),
                embeddings.get(i + 1)
            );
            distances.add(distance);
        }

        // 4. 在语义距离大的地方分割
        List<String> chunks = new ArrayList<>();
        StringBuilder currentChunk = new StringBuilder();

        // 找出语义距离的阈值（如前25%分位数）
        double threshold = calculateThreshold(distances, 0.75);

        for (int i = 0; i < paragraphs.size(); i++) {

            currentChunk.append(paragraphs.get(i)).append("\n\n");

            // 如果是最后一段，或与下一段语义差异大
            if (i == paragraphs.size() - 1 ||
                distances.get(i) > threshold) {

                chunks.add(currentChunk.toString().trim());
                currentChunk = new StringBuilder();
            }
        }

        return chunks;
    }

    private double calculateThreshold(List<Double> distances,
                                     double percentile) {
        // 计算分位数作为阈值
        List<Double> sorted = new ArrayList<>(distances);
        Collections.sort(sorted);

        int index = (int) (sorted.size() * percentile);
        return sorted.get(index);
    }
}

优点：

  • 语义完整性最好
  • 检索质量最高

缺点：

  • 计算成本高（需要多次Embedding）
  • 实现复杂

策略4：递归分块（Recursive）

@Service
public class RecursiveChunker {

    // 多级分割符：从粗到细
    private static final String[] SEPARATORS = {
        "\n\n\n",  // 三次换行（章节）
        "\n\n",    // 两次换行（段落）
        "\n",      // 一次换行（行）
        "。",      // 句号（句子）
        "，",      // 逗号（短语）
        " "        // 空格（词）
    };

    public List<String> chunkRecursive(String text,
                                       int chunkSize,
                                       int overlap) {

        return chunkRecursive(text, chunkSize, overlap, 0);
    }

    private List<String> chunkRecursive(String text,
                                       int chunkSize,
                                       int overlap,
                                       int separatorIndex) {

        List<String> chunks = new ArrayList<>();

        // 如果文本足够短，直接返回
        if (text.length() <= chunkSize) {
            chunks.add(text);
            return chunks;
        }

        // 使用当前分割符
        if (separatorIndex < SEPARATORS.length) {

            String separator = SEPARATORS[separatorIndex];
            String[] splits = text.split(Pattern.quote(separator));

            StringBuilder currentChunk = new StringBuilder();

            for (String split : splits) {

                String trimmed = split.trim();

                // 如果单个split太长，递归使用更细的分割符
                if (trimmed.length() > chunkSize) {
                    if (currentChunk.length() > 0) {
                        chunks.add(currentChunk.toString());
                        currentChunk = new StringBuilder();
                    }

                    chunks.addAll(chunkRecursive(
                        trimmed,
                        chunkSize,
                        overlap,
                        separatorIndex + 1
                    ));
                    continue;
                }

                // 检查是否超出chunk大小
                if (currentChunk.length() + trimmed.length() + separator.length()
                    > chunkSize) {

                    if (currentChunk.length() > 0) {
                        chunks.add(currentChunk.toString());
                        currentChunk = new StringBuilder();
                    }
                }

                // 添加到当前chunk
                if (currentChunk.length() > 0) {
                    currentChunk.append(separator);
                }
                currentChunk.append(trimmed);
            }

            // 添加最后一个chunk
            if (currentChunk.length() > 0) {
                chunks.add(currentChunk.toString());
            }

        } else {
            // 没有更细的分割符，强制分割
            chunks.addAll(forceChunk(text, chunkSize, overlap));
        }

        return chunks;
    }

    private List<String> forceChunk(String text,
                                    int chunkSize,
                                    int overlap) {

        List<String> chunks = new ArrayList<>();
        int start = 0;

        while (start < text.length()) {
            int end = Math.min(start + chunkSize, text.length());
            chunks.add(text.substring(start, end));
            start = end - overlap;
        }

        return chunks;
    }
}

优点：

  • 灵活性高
  • 保持语义结构
  • 适用各种文本

缺点：

  • 实现复杂
  • 需要调参

最佳实践

@Service
public class ChunkingBestPractices {

    // 推荐配置
    private static final int OPTIMAL_CHUNK_SIZE = 500;  // 500-1000字符
    private static final int OPTIMAL_OVERLAP = 50;      // 10-20%重叠

    // 实际应用：综合策略
    public List<Chunk> smartChunk(String document) {

        // 1. 预处理
        String cleaned = cleanText(document);

        // 2. 按结构分割（章节、段落）
        List<String> structuralChunks = splitByStructure(cleaned);

        // 3. 每个结构块内部递归分块
        List<Chunk> chunks = new ArrayList<>();
        for (String structuralChunk : structuralChunks) {

            List<String> subChunks = chunkRecursive(
                structuralChunk,
                OPTIMAL_CHUNK_SIZE,
                OPTIMAL_OVERLAP
            );

            // 4. 为每个chunk添加元数据
            for (int i = 0; i < subChunks.size(); i++) {
                chunks.add(new Chunk(
                    subChunks.get(i),
                    buildMetadata(structuralChunk, i, subChunks.size())
                ));
            }
        }

        return chunks;
    }

    private String cleanText(String text) {
        // 移除多余空白
        text = text.replaceAll("\\s+", " ");

        // 修复乱码
        text = fixEncoding(text);

        // 统一换行符
        text = text.replaceAll("\\r\\n|\\r", "\n");

        return text;
    }

    private Map<String, Object> buildMetadata(String source,
                                              int index,
                                              int total) {

        Map<String, Object> metadata = new HashMap<>();
        metadata.put("source", source);
        metadata.put("chunk_index", index);
        metadata.put("total_chunks", total);
        metadata.put("timestamp", Instant.now());

        return metadata;
    }
}

Chunk Size选择

推荐配置（经验值）：

短文本（< 1000字）：
├─ Chunk Size: 200-300 字符
├─ Overlap: 20-50 字符
└─ 场景：FAQ、对话

中等文本（1000-5000字）：
├─ Chunk Size: 500-800 字符
├─ Overlap: 50-100 字符
└─ 场景：文章、报告

长文本（> 5000字）：
├─ Chunk Size: 800-1000 字符
├─ Overlap: 100-150 字符
└─ 场景：书籍、文档

技术文档：
├─ Chunk Size: 按代码块分割
├─ Overlap: 保留上下文
└─ 场景：API文档、代码

2. Data Cleaning（数据清洗）

🎯 掌握程度：必须掌握

什么是数据清洗？

通俗解释：
数据清洗就像"打扫房间"。把脏乱的数据整理干净，让知识库更整洁、更准确。

技术定义：
去除数据中的噪声、错误、重复等内容，提升数据质量的过程。

为什么需要数据清洗？

脏数据的问题：
❌ 影响检索质量：垃圾进，垃圾出
❌ 浪费存储空间：重复数据占用资源
❌ 降低模型性能：噪声干扰语义理解
❌ 误导AI回答：错误信息产生错误答案

✅ 清洗后的好处：
├─ 提升检索精度：相关性更高
├─ 节省存储成本：去重减少冗余
├─ 提高响应速度：数据量更小
└─ 保证回答质量：信息准确可靠

常见数据问题

1. 格式问题
   ├─ 混乱换行：\r\n \n \r 不统一
   ├─ 特殊字符：乱码、控制字符
   ├─ 空白过多：连续空格、空行
   └─ 编码问题：UTF-8 GBK 混乱

2. 内容问题
   ├─ 重复内容：完全相同或高度相似
   ├─ 无意义内容：广告、导航栏、版权声明
   ├─ 过时信息：失效的链接、过期的数据
   └─ 错误信息：拼写错误、逻辑矛盾

3. 结构问题
   ├─ 缺失字段：关键字段为空
   ├─ 格式不一致：日期、数字格式混乱
   ├─ 嵌套过深：HTML标签嵌套复杂
   └─ 编码特殊字符：HTML实体未解码

数据清洗流程

@Service
public class DataCleaningPipeline {

    public String clean(String rawText) {

        // 1. 基础清洗
        String text = basicClean(rawText);

        // 2. 深度清洗
        text = deepClean(text);

        // 3. 结构化处理
        text = structureClean(text);

        // 4. 验证
        validate(text);

        return text;
    }

    private String basicClean(String text) {
        // 统一换行符
        text = text.replaceAll("\\r\\n|\\r", "\n");

        // 移除控制字符
        text = text.replaceAll("[\\p{Cntrl}&&[^\n\t]]", "");

        // 移除多余空白
        text = text.replaceAll(" +", " ");
        text = text.replaceAll("\n{3,}", "\n\n");

        // 修复编码
        text = fixEncoding(text);

        return text;
    }

    private String deepClean(String text) {
        // 移除页眉页脚
        text = removeHeadersFooters(text);

        // 移除版权声明
        text = removeCopyright(text);

        // 移除广告内容
        text = removeAds(text);

        // 移除无意义内容
        text = removeNoise(text);

        return text;
    }

    private String structureClean(String text) {
        // 解码HTML实体
        text = HtmlUtils.htmlUnescape(text);

        // 标准化日期格式
        text = normalizeDates(text);

        // 标准化数字格式
        text = normalizeNumbers(text);

        return text;
    }

    private void validate(String text) {
        // 检查是否为空
        if (text.trim().isEmpty()) {
            throw new IllegalArgumentException("清洗后文本为空");
        }

        // 检查长度
        if (text.length() < 50) {
            throw new IllegalArgumentException("文本过短");
        }

        // 检查是否包含有效内容
        if (!hasValidContent(text)) {
            throw new IllegalArgumentException("文本无有效内容");
        }
    }
}

高级清洗技巧

@Service
public class AdvancedDataCleaning {

    // 智能去重
    public String deduplicate(String text) {
        // 1. 计算文本指纹（SimHash）
        String fingerprint = simHash(text);

        // 2. 查找相似文档
        List<String> similarDocs = findSimilar(fingerprint);

        // 3. 如果存在高度相似文档，返回已存在的
        if (hasHighlySimilar(similarDocs, text)) {
            return null;  // 表示重复，不保存
        }

        return text;
    }

    // 语言检测
    public String filterByLanguage(String text) {
        // 检测文本语言
        Language language = detectLanguage(text);

        // 只保留中文和英文
        if (!language.equals(Language.CHINESE) &&
            !language.equals(Language.ENGLISH)) {
            return null;
        }

        return text;
    }

    // 内容质量评分
    public double scoreQuality(String text) {
        double score = 0.0;

        // 1. 长度评分（适中为佳）
        int length = text.length();
        if (length >= 200 && length <= 5000) {
            score += 0.3;
        }

        // 2. 结构评分（有段落结构）
        if (text.contains("\n\n")) {
            score += 0.2;
        }

        // 3. 内容评分（包含关键词）
        if (containsKeywords(text)) {
            score += 0.3;
        }

        // 4. 完整性评分（句子完整）
        if (hasCompleteSentences(text)) {
            score += 0.2;
        }

        return score;
    }

    // 移除敏感信息
    public String removeSensitiveInfo(String text) {
        // 移除邮箱
        text = text.replaceAll("\\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Z|a-z]{2,}\\b", "[邮箱]");

        // 移除手机号
        text = text.replaceAll("\\b1[3-9]\\d{9}\\b", "[手机号]");

        // 移除身份证号
        text = text.replaceAll("\\b\\d{17}[\\dXx]\\b", "[身份证]");

        // 移除银行卡号
        text = text.replaceAll("\\b\\d{16,19}\\b", "[银行卡]");

        return text;
    }
}

💡 数据清洗建议

  • 渐进式清洗：先做基础清洗，再做深度清洗，每步都验证结果
  • 保留原始数据：清洗前备份原始数据，以便回溯和调试
  • 可配置化：将清洗规则配置化，便于调整和维护
  • 质量监控：建立数据质量监控机制，及时发现和修复问题
  • A/B测试：对比不同清洗策略的效果，选择最优方案
推荐清洗优先级：
1️⃣ 高优先级（必须清洗）：
   ├─ 统一编码和换行符
   ├─ 移除控制字符
   ├─ 去除完全重复的内容
   └─ 移除敏感信息

2️⃣ 中优先级（建议清洗）：
   ├─ 移除页眉页脚
   ├─ 移除广告和导航
   ├─ 去除高度相似内容
   └─ 修复拼写错误

3️⃣ 低优先级（可选清洗）：
   ├─ 内容质量评分
   ├─ 语言过滤
   ├─ 过时信息清理
   └─ 格式美化

3. Metadata Management（元数据管理）

🎯 掌握程度：必须掌握

什么是元数据？

通俗解释：
元数据就是"数据的标签"。就像商品的价格标签，告诉你这个商品叫什么、多少钱、哪里生产的。元数据告诉你文档是谁写的、什么时候写的、属于哪个类别。

技术定义：
描述数据属性的信息，包括数据的来源、结构、特征、关系等。

为什么需要元数据？

元数据的作用：
📚 数据管理
├─ 追踪来源：知道数据从哪里来
├─ 版本控制：管理数据的不同版本
├─ 分类组织：按类别组织数据
└─ 权限控制：控制谁可以访问

🔍 检索增强
├─ 精确过滤：按元数据筛选结果
├─ 上下文补充：给AI提供更多背景信息
├─ 结果排序：按相关性、时间等排序
└─ 结果去重：识别重复内容

💪 智能应用
├─ 个性化推荐：基于用户偏好
├─ 权限控制：基于角色访问
├─ 审计追踪：记录数据使用情况
└─ 数据分析：分析数据使用模式

常见元数据类型

1. 基础元数据
   ├─ 文档ID：唯一标识符
   ├─ 标题：文档标题
   ├─ 摘要：内容摘要
   ├─ 关键词：内容关键词
   └─ 语言：文档语言

2. 来源元数据
   ├─ 来源URL：原始网址
   ├─ 来源类型：网页/文件/数据库
   ├─ 作者：作者名称
   ├─ 创建时间：文档创建日期
   └─ 更新时间：最后更新日期

3. 内容元数据
   ├─ 字数：文档长度
   ├─ 段落数：结构信息
   ├─ 分类：内容类别
   ├─ 标签：内容标签
   └─ 相似文档：相关文档ID

4. 技术元数据
   ├─ Chunk索引：当前chunk索引
   ├─ 总Chunk数：文档总chunk数
   ├─ 向量模型：使用的embedding模型
   ├─ Token数：token使用量
   └─ 处理时间：处理耗时

5. 业务元数据
   ├─ 部门：所属部门
   ├─ 项目：所属项目
   ├─ 优先级：重要程度
   ├─ 访问权限：谁可以访问
   └─ 有效期：数据有效期

@Service
public class MetadataManager {

    // 创建文档元数据
    public DocumentMetadata createDocumentMetadata(
        String content,
        String sourceUrl,
        String author) {

        DocumentMetadata metadata = new DocumentMetadata();

        // 基础信息
        metadata.setId(UUID.randomUUID().toString());
        metadata.setTitle(extractTitle(content));
        metadata.setSummary(extractSummary(content));

        // 来源信息
        metadata.setSourceUrl(sourceUrl);
        metadata.setSourceType("web");
        metadata.setAuthor(author);
        metadata.setCreatedAt(Instant.now());

        // 内容分析
        metadata.setWordCount(countWords(content));
        metadata.setLanguage(detectLanguage(content));
        metadata.setKeywords(extractKeywords(content));

        // 技术信息
        metadata.setVectorModel("text-embedding-ada-002");
        metadata.setEstimatedTokens(estimateTokens(content));

        return metadata;
    }

    // 创建Chunk元数据
    public ChunkMetadata createChunkMetadata(
        DocumentMetadata docMetadata,
        String chunkContent,
        int chunkIndex,
        int totalChunks) {

        ChunkMetadata metadata = new ChunkMetadata();

        // 继承文档元数据
        metadata.setDocumentId(docMetadata.getId());
        metadata.setTitle(docMetadata.getTitle());
        metadata.setSourceUrl(docMetadata.getSourceUrl());
        metadata.setAuthor(docMetadata.getAuthor());

        // Chunk特有信息
        metadata.setChunkIndex(chunkIndex);
        metadata.setTotalChunks(totalChunks);
        metadata.setContentLength(chunkContent.length());
        metadata.setCreatedAt(Instant.now());

        // 计算chunk的位置信息
        metadata.setStartPosition(calculateStartPosition(chunkIndex, chunkContent.length()));
        metadata.setEndPosition(metadata.getStartPosition() + chunkContent.length());

        return metadata;
    }

    // 提取标题
    private String extractTitle(String content) {
        // 尝试从内容中提取标题（如第一个#标题）
        Pattern pattern = Pattern.compile("^#\\s+(.+)$", Pattern.MULTILINE);
        Matcher matcher = pattern.matcher(content);

        if (matcher.find()) {
            return matcher.group(1).trim();
        }

        // 如果没有找到，使用前50个字符作为标题
        return content.substring(0, Math.min(50, content.length())) + "...";
    }

    // 提取摘要
    private String extractSummary(String content) {
        // 移除标题和空行
        String[] lines = content.split("\n");
        StringBuilder summary = new StringBuilder();

        for (String line : lines) {
            line = line.trim();
            if (!line.isEmpty() && !line.startsWith("#")) {
                summary.append(line).append(" ");
                if (summary.length() >= 200) {
                    break;
                }
            }
        }

        return summary.toString().trim();
    }

    // 提取关键词
    private List<String> extractKeywords(String content) {
        // 使用TF-IDF或RAKE算法提取关键词
        // 这里简化处理，使用词频统计
        Map<String, Integer> wordFreq = new HashMap<>();

        // 分词（简化版）
        String[] words = content.split("[\\s\\p{Punct}]+");

        for (String word : words) {
            if (word.length() >= 2) {
                wordFreq.merge(word, 1, Integer::sum);
            }
        }

        // 返回词频最高的10个词
        return wordFreq.entrySet().stream()
            .sorted(Map.Entry.<String, Integer>comparingByValue().reversed())
            .limit(10)
            .map(Map.Entry::getKey)
            .collect(Collectors.toList());
    }
}

4. Re-ranking（重排序）

🎯 掌握程度：必须掌握

什么是Re-ranking？

通俗解释：
Re-ranking就是"二次筛选"。第一次检索可能返回很多结果，但不够精确。重排序就像老师批改作业，把最相关的答案挑出来，排在最前面。

技术定义：
对初始检索结果进行重新排序，使用更精确的模型计算相关性，提升结果质量。

为什么需要Re-ranking？

传统检索的问题：
❌ 向量检索不够精确
   ├─ 语义相似度计算有误差
   ├─ 可能返回不相关的内容
   └─ 无法处理复杂的查询需求

❌ 简单的相似度计算
   ├─ 只考虑整体语义
   ├─ 忽略具体的关键词匹配
   └─ 无法理解查询意图

✅ Re-ranking的优势：
├─ 精确度高：使用更强的模型
├─ 结果更好：最相关的内容排在前面
├─ 灵活性强：可以自定义排序规则
└─ 成本可控：只对少量结果重排序

方式1：Cross-encoder重排序

@Service
public class CrossEncoderReRanker {

    @Autowired
    private CrossEncoderModel crossEncoder;

    public List<Document> reRank(
        String query,
        List<Document> candidates,
        int topK) {

        // 1. 计算query与每个候选文档的相关性分数
        List<ScoreDocument> scoredDocs = new ArrayList<>();

        for (Document doc : candidates) {
            // 使用Cross-encoder计算精确相关性
            double score = crossEncoder.calculateScore(
                query,
                doc.getContent()
            );

            scoredDocs.add(new ScoreDocument(doc, score));
        }

        // 2. 按分数排序
        scoredDocs.sort((a, b) -> Double.compare(b.score, a.score));

        // 3. 返回Top K结果
        return scoredDocs.stream()
            .limit(topK)
            .map(sd -> sd.document)
            .collect(Collectors.toList());
    }
}

方式2：基于特征的重排序

@Service
public class FeatureBasedReRanker {

    public List<Document> reRank(String query, List<Document> candidates) {

        List<ScoreDocument> scoredDocs = new ArrayList<>();

        for (Document doc : candidates) {
            double score = calculateCompositeScore(query, doc);
            scoredDocs.add(new ScoreDocument(doc, score));
        }

        // 按综合分数排序
        scoredDocs.sort((a, b) -> Double.compare(b.score, a.score));

        return scoredDocs.stream()
            .map(sd -> sd.document)
            .collect(Collectors.toList());
    }

    private double calculateCompositeScore(String query, Document doc) {
        double score = 0.0;

        // 1. 向量相似度（权重：0.4）
        double vectorSim = calculateVectorSimilarity(query, doc);
        score += vectorSim * 0.4;

        // 2. 关键词匹配度（权重：0.3）
        double keywordMatch = calculateKeywordMatch(query, doc);
        score += keywordMatch * 0.3;

        // 3. BM25分数（权重：0.2）
        double bm25Score = calculateBM25(query, doc);
        score += bm25Score * 0.2;

        // 4. 文档新鲜度（权重：0.1）
        double freshness = calculateFreshness(doc);
        score += freshness * 0.1;

        return score;
    }

    private double calculateKeywordMatch(String query, Document doc) {
        Set<String> queryKeywords = extractKeywords(query);
        String content = doc.getContent().toLowerCase();

        long matchCount = queryKeywords.stream()
            .filter(keyword -> content.contains(keyword.toLowerCase()))
            .count();

        return (double) matchCount / queryKeywords.size();
    }

    private double calculateFreshness(Document doc) {
        LocalDate docDate = (LocalDate) doc.getMetadata().get("createdAt");
        LocalDate now = LocalDate.now();

        long daysOld = ChronoUnit.DAYS.between(docDate, now);

        // 越新的文档分数越高
        return Math.max(0, 1.0 - daysOld / 365.0);
    }
}

5. Hybrid Search（混合检索）

🎯 掌握程度：必须掌握

什么是混合检索？

通俗解释：
混合检索就是"多管齐下"。同时用多种方法找答案，比如既查意思相似的，又查关键词完全匹配的，然后把结果合并，取长补短。

技术定义：
结合多种检索技术（如向量检索、关键词检索、BM25等），综合多种信号进行检索，提升召回率和准确率。

为什么需要混合检索？

单一检索的局限性：
❌ 纯向量检索
   ├─ 可能漏掉精确匹配的关键词
   ├─ 对专业术语、缩写不敏感
   └─ 无法处理字面匹配的需求

❌ 纯关键词检索
   ├─ 无法理解语义相似性
   ├─ 需要精确的关键词匹配
   └─ 对同义词、变体不敏感

✅ 混合检索的优势：
├─ 召回率更高：结合多种检索方式
├─ 准确率更高：互补优势
├─ 鲁棒性更强：不依赖单一方法
└─ 灵活性更高：可调整不同方法的权重

混合检索实现

@Service
public class HybridSearchService {

    @Autowired
    private VectorSearchService vectorSearch;

    @Autowired
    private KeywordSearchService keywordSearch;

    @Autowired
    private ReRankingService reRankingService;

    public List<Document> hybridSearch(
        String query,
        int topK) {

        // 1. 并行检索
        CompletableFuture<List<Document>> vectorFuture =
            CompletableFuture.supplyAsync(() -> vectorSearch.search(query, 100));

        CompletableFuture<List<Document>> keywordFuture =
            CompletableFuture.supplyAsync(() -> keywordSearch.search(query, 100));

        // 等待两个检索完成
        CompletableFuture.allOf(vectorFuture, keywordFuture).join();

        List<Document> vectorResults = vectorFuture.join();
        List<Document> keywordResults = keywordFuture.join();

        // 2. 合并结果
        List<Document> mergedResults = mergeResults(
            vectorResults,
            keywordResults
        );

        // 3. 重排序
        List<Document> reRankResults = reRankingService.reRank(
            query,
            mergedResults,
            topK
        );

        return reRankResults;
    }

    private List<Document> mergeResults(
        List<Document> vectorResults,
        List<Document> keywordResults) {

        Map<String, MergedDocument> mergedMap = new LinkedHashMap<>();

        // 处理向量检索结果
        for (int i = 0; i < vectorResults.size(); i++) {
            Document doc = vectorResults.get(i);
            String docId = doc.getId();

            // 计算分数（排名越前分数越高）
            double score = 1.0 / (i + 1);

            MergedDocument merged = new MergedDocument(doc);
            merged.setVectorScore(score);
            mergedMap.put(docId, merged);
        }

        // 处理关键词检索结果
        for (int i = 0; i < keywordResults.size(); i++) {
            Document doc = keywordResults.get(i);
            String docId = doc.getId();

            double score = 1.0 / (i + 1);

            if (mergedMap.containsKey(docId)) {
                // 文档已存在，累加分数
                mergedMap.get(docId).setKeywordScore(score);
            } else {
                // 新文档
                MergedDocument merged = new MergedDocument(doc);
                merged.setKeywordScore(score);
                mergedMap.put(docId, merged);
            }
        }

        // 计算综合分数并排序
        return mergedMap.values().stream()
            .peek(merged -> {
                // 加权融合：向量0.6 + 关键词0.4
                double compositeScore =
                    merged.getVectorScore() * 0.6 +
                    merged.getKeywordScore() * 0.4;
                merged.setCompositeScore(compositeScore);
            })
            .sorted((a, b) -> Double.compare(
                b.getCompositeScore(),
                a.getCompositeScore()
            ))
            .map(MergedDocument::getDocument)
            .collect(Collectors.toList());
    }
}

// 合并的文档对象
@Data
private static class MergedDocument {
    private Document document;
    private double vectorScore;
    private double keywordScore;
    private double compositeScore;

    public MergedDocument(Document document) {
        this.document = document;
        this.vectorScore = 0.0;
        this.keywordScore = 0.0;
        this.compositeScore = 0.0;
    }
}

// Reciprocal Rank Fusion
// 公式：score = Σ 1/(k + rank_i)
// k是常数（通常k=60）

public double calculateRRF(
    List<Integer> ranks,
    int k) {

    double score = 0.0;
    for (int rank : ranks) {
        score += 1.0 / (k + rank);
    }
    return score;
}

// 示例：
// 文档A：向量检索rank=1，关键词检索rank=5
// score = 1/(60+1) + 1/(60+5) = 0.0164 + 0.0154 = 0.0318

        scoredDocs.sort((a, b) -> Double.compare(b.score, a.score));

        return scoredDocs.stream()
            .map(sd -> sd.document)
            .collect(Collectors.toList());
    }

    private Map<String, Double> extractFeatures(
        String query,
        Document doc) {

        Map<String, Double> features = new HashMap<>();

        // 特征1：向量相似度
        features.put("vector_sim", calculateVectorSimilarity(query, doc));

        // 特征2：关键词匹配度
        features.put("keyword_match", calculateKeywordMatch(query, doc));

        // 特征3：BM25分数
        features.put("bm25_score", calculateBM25(query, doc));

        // 特征4：文档长度
        features.put("doc_length", (double) doc.getContent().length());

        // 特征5：文档新鲜度
        features.put("freshness", calculateFreshness(doc));

        // 特征6：查询长度
        features.put("query_length", (double) query.length());

        return features;
    }
}

@Service
public class KnowledgeBaseBuilder {

    @Autowired
    private DataLoader dataLoader;

    @Autowired
    private DataCleaner dataCleaner;

    @Autowired
    private Chunker chunker;

    @Autowired
    private EmbeddingService embeddingService;

    @Autowired
    private VectorStore vectorStore;

    @Autowired
    private MetadataExtractor metadataExtractor;

    // 构建知识库
    public BuildResult buildKnowledgeBase(BuildConfig config) {

        List<String> errors = new ArrayList<>();
        int successCount = 0;
        int skipCount = 0;

        try {
            // 1. 加载数据
            log.info("开始加载数据...");
            List<Document> documents = dataLoader.load(config.getDataPaths());
            log.info("加载完成，共{}个文档", documents.size());

            // 2. 处理每个文档
            for (Document doc : documents) {
                try {
                    processDocument(doc, config);
                    successCount++;

                    if (successCount % 100 == 0) {
                        log.info("已处理{}个文档", successCount);
                    }

                } catch (SkipException e) {
                    skipCount++;
                    log.warn("跳过文档：{} - {}", doc.getId(), e.getMessage());

                } catch (Exception e) {
                    errors.add(doc.getId() + ": " + e.getMessage());
                    log.error("处理文档失败：{}", doc.getId(), e);
                }
            }

            // 3. 保存构建统计
            BuildResult result = new BuildResult();
            result.setTotalDocuments(documents.size());
            result.setSuccessCount(successCount);
            result.setSkipCount(skipCount);
            result.setErrorCount(errors.size());
            result.setErrors(errors);

            log.info("知识库构建完成：成功={}, 跳过={}, 失败={}",
                successCount, skipCount, errors.size());

            return result;

        } catch (Exception e) {
            log.error("知识库构建失败", e);
            throw new BuildException("知识库构建失败", e);
        }
    }

    // 处理单个文档
    private void processDocument(Document doc, BuildConfig config) {

        // 1. 数据清洗
        String cleanedContent = dataCleaner.clean(doc.getContent());

        // 验证清洗后的内容
        if (cleanedContent.length() < config.getMinDocLength()) {
            throw new SkipException("文档过短");
        }

        // 2. 提取元数据
        DocumentMetadata metadata = metadataExtractor.extract(doc, cleanedContent);

        // 3. 文档分块
        List<Chunk> chunks = chunker.chunk(cleanedContent, config.getChunkConfig());

        // 验证分块结果
        if (chunks.isEmpty()) {
            throw new SkipException("分块结果为空");
        }

        // 4. 向量化并存储
        List<VectorChunk> vectorChunks = new ArrayList<>();

        for (int i = 0; i < chunks.size(); i++) {
            Chunk chunk = chunks.get(i);

            // 创建chunk元数据
            ChunkMetadata chunkMetadata = createChunkMetadata(
                metadata,
                chunk,
                i,
                chunks.size()
            );

            // 向量化
            float[] embedding = embeddingService.embed(chunk.getContent());

            // 创建向量chunk
            VectorChunk vectorChunk = new VectorChunk(
                chunk.getId(),
                embedding,
                chunk.getContent(),
                chunkMetadata
            );

            vectorChunks.add(vectorChunk);
        }

        // 5. 批量存储到向量数据库
        vectorStore.add(vectorChunks);

        log.debug("文档处理成功：{}, chunk数量：{}", doc.getId(), chunks.size());
    }

    private ChunkMetadata createChunkMetadata(
        DocumentMetadata docMetadata,
        Chunk chunk,
        int index,
        int total) {

        ChunkMetadata metadata = new ChunkMetadata();

        // 继承文档元数据
        metadata.setDocumentId(docMetadata.getId());
        metadata.setTitle(docMetadata.getTitle());
        metadata.setSourceUrl(docMetadata.getSourceUrl());
        metadata.setAuthor(docMetadata.getAuthor());
        metadata.setDepartment(docMetadata.getDepartment());

        // Chunk特有元数据
        metadata.setChunkIndex(index);
        metadata.setTotalChunks(total);
        metadata.setContentLength(chunk.getContent().length());
        metadata.setCreatedAt(Instant.now());

        // 添加位置信息
        metadata.setStartPosition(chunk.getStartPosition());
        metadata.setEndPosition(chunk.getEndPosition());

        return metadata;
    }
}

增量更新策略

@Service
public class KnowledgeBaseUpdater {

    @Autowired
    private VectorStore vectorStore;

    @Autowired
    private DocumentRepository docRepository;

    // 增量更新知识库
    public void updateKnowledgeBase(List<Document> newDocuments) {

        for (Document doc : newDocuments) {

            // 1. 检查文档是否已存在
            Optional<Document> existing = docRepository.findById(doc.getId());

            if (existing.isPresent()) {
                // 文档已存在，检查是否需要更新
                if (shouldUpdate(existing.get(), doc)) {
                    // 删除旧的chunks
                    deleteDocumentChunks(doc.getId());

                    // 处理新文档
                    processDocument(doc);

                    log.info("文档已更新：{}", doc.getId());
                } else {
                    log.debug("文档无需更新：{}", doc.getId());
                }
            } else {
                // 新文档，直接处理
                processDocument(doc);
                log.info("新文档已添加：{}", doc.getId());
            }
        }
    }

    // 判断是否需要更新
    private boolean shouldUpdate(Document existing, Document newDoc) {

        // 1. 检查内容是否变化
        if (!existing.getContent().equals(newDoc.getContent())) {
            return true;
        }

        // 2. 检查更新时间
        if (newDoc.getUpdatedAt().isAfter(existing.getUpdatedAt())) {
            return true;
        }

        // 3. 检查版本号
        if (newDoc.getVersion() > existing.getVersion()) {
            return true;
        }

        return false;
    }

    // 删除文档的所有chunks
    private void deleteDocumentChunks(String documentId) {

        // 构建删除条件
        DeleteQuery query = DeleteQuery.builder()
            .filter(Expression.builder()
                .key("document_id")
                .value(documentId)
                .build())
            .build();

        // 执行删除
        vectorStore.delete(query);

        log.info("已删除文档的chunks：{}", documentId);
    }
}
